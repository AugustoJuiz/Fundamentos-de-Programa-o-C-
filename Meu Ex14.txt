int main()
{

    int c7 = 0;

    //CPF
    for(j=0; j<11; j++)
    {
        //vet
        for(i=0; i<10; i++)
        {
            if(CPF[j]==vet[i])
            {
                if(i==0)
                {
                    c0++;
                }

                {
                    c1++;
                }
                else if (i==2)

                }

                {

                }
                else if (i==8)
                {
                    c8++;
                }

#include <stdio.h>

int main() 
{
    long long int cpf;
    int cont[10] = {0};
    int digito;
    int i;

    scanf("%lld", &cpf);

    while (cpf != 0) 
    {
        digito = cpf%10;
        cont[digito]++;
        cpf = cpf/10;
    }

    for (i = 0; i < 10; i++) 
    {
        printf("%d = %d\n", i, cont[i]);
    }

    return 0;
}
