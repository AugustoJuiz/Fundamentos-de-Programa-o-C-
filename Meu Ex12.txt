Faça um programa que leia uma frase (string) contendo 4 valores numéricos reais de precisão dupla (double), separados por espaços. A seguir, seu programa deve separar cada número dos demais, convertê-lo para o formato numérico e exibir a soma dos números.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main()
{
    char n1[5];
    char n2[5];
    char n3[5];
    char n4[5];
    
    scanf("%s", n1);
    scanf("%s", n2);
    scanf("%s", n3);
    scanf("%s", n4);
    
    double n1_double = atof(n1);
    double n2_double = atof(n2);
    double n3_double = atof(n3);
    double n4_double = atof(n4);
    double soma;
    
    soma = n1_double + n2_double + n3_double + n4_double;
    
    printf("%lf", soma);
    
    return 0;
}
